# Python Implementation
def detect_and_classify(frame, mask):
    # Preprocessing
    processed, thresholded = preprocess(frame, mask)
    
    # Find contours
    contours, _ = cv.findContours(thresholded, 
                                   cv2.RETR_EXTERNAL, 
                                   cv2.CHAIN_APPROX_SIMPLE)
    
    # Count valid contours
    valid_contours = 0
    for c in contours:
        area = cv.contourArea(c)
        if 50 <= area <= 1000:
            perimeter = cv.arcLength(c, True)
            if perimeter > 0:
                circularity = 4 * np.pi * area / (perimeter * perimeter)
                if 0.3 <= circularity <= 1.0:
                    valid_contours += 1
    
    # Simple classification
    if valid_contours == 0:
        return "ISOTROPIC", valid_contours
    else:
        return "CHOLESTERIC", valid_contours

function processSingleFrame(canvasElement, frameName, timestamp) {
    // Read image and convert to grayscale
    let src = cv.imread(canvasElement);
    let gray = new cv.Mat();
    cv.cvtColor(src, gray, cv.COLOR_RGBA2GRAY);
    
    // Create mask and apply preprocessing
    let mask = createElectrodeMask(gray);
    let processed = preprocessImage(src, mask);
    
    // Threshold and find contours
    let thresh = new cv.Mat();
    cv.adaptiveThreshold(processed, thresh, 255,
                         cv.ADAPTIVE_THRESH_GAUSSIAN_C,
                         cv.THRESH_BINARY_INV, 11, 2);
    
    // Return classification result
    return {
        name: frameName,
        timestamp: timestamp,
        validContours: countValidContours(thresh),
        phase: classifyPhase(countValidContours(thresh))
    };
}
// Simple classification based on contours
function classifyPhase(contourCount) {
    /*
    Binary classification rule:
    - If contour count = 0 ‚Üí ISOTROPIC phase
    - If contour count > 0 ‚Üí CHOLESTERIC phase
    
    This matches the Python implementation:
    if num_contours == 0:
        phase = "ISOTROPIC"
    else:
        phase = "CHOLESTERIC"
    */
    return contourCount === 0 ? "ISOTROPIC" : "CHOLESTERIC";
}
